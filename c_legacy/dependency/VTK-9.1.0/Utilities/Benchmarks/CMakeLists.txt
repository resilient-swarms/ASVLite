    PRIVATE
      VTK::UtilitiesBenchmarks)

    TARGETS TimingTests
    MODULES VTK::UtilitiesBenchmarks)

  vtk_module_add_executable(GLBenchmarking
    NO_INSTALL
    GLBenchmarking.cxx)
  target_link_libraries(GLBenchmarking
    PRIVATE
      VTK::ChartsCore
      VTK::UtilitiesBenchmarks
      VTK::ViewsContext2D)

  vtk_module_autoinit(
    TARGETS GLBenchmarking
    MODULES VTK::ChartsCore
            VTK::UtilitiesBenchmarks
